} loadAdminStats(); loadUserManagement(); loadPatientAssignment(); loadSystemReports(); setupDashboardNavigation(); setupEventListeners(); } function setupEventListeners() { document.getElementById('logout-btn').addEventListener('click', function() { localStorage.removeItem('currentUser'); window.location.href = 'index.html'; }); const addUserBtn = document.getElementById('add-user-btn'); if (addUserBtn) { addUserBtn.addEventListener('click', showAddUserModal); } const assignBtn = document.getElementById('assign-btn'); if (assignBtn) { assignBtn.addEventListener('click', assignPatient); } } function setupDashboardNavigation() { document.querySelectorAll('.dashboard-link').forEach(link => { link.addEventListener('click', function(e) { e.preventDefault(); document.querySelectorAll('.dashboard-link').forEach(l => l.classList.remove('active')); this.classList.add('active'); loadSectionData(sectionId); }); }); } function loadSectionData(sectionId) { switch(sectionId) { case 'stats': loadAdminStats(); break; case 'user-management': loadUserManagement(); break; case 'patient-assignment': loadPatientAssignment(); break; case 'system-reports': loadSystemReports(); break; } } function loadAdminStats() { const allUsers = JSON.parse(localStorage.getItem('allUsers') || '[]'); const totalUsers = allUsers.length; const therapists = allUsers.filter(u => u.role === 'therapist').length; const patients = allUsers.filter(u => u.role === 'user').length; const admins = allUsers.filter(u => u.role === 'admin').length; const statsGrid = document.getElementById('stats-grid'); if (!statsGrid) return; statsGrid.innerHTML = ''; const stats = [ { title: 'Total Users', value: totalUsers, icon: '👥', color: '#4a6fa5' }, { title: 'Therapists', value: therapists, icon: '👨‍⚕️', color: '#6b8cbc' }, { title: 'Patients', value: patients, icon: '😊', color: '#28a745' }, { title: 'Admins', value: admins, icon: '🛡️', color: '#ff7e5f' } ]; stats.forEach(stat => { const statCard = document.createElement('div'); statCard.className = 'stat-card'; statCard.innerHTML = `
${stat.icon}
${stat.value}
${stat.title}
`; statsGrid.appendChild(statCard); }); } function loadUserManagement() { const allUsers = JSON.parse(localStorage.getItem('allUsers') || '[]'); const usersList = document.getElementById('users-list'); if (!usersList) return; usersList.innerHTML = ''; if (allUsers.length === 0) { usersList.innerHTML = '
No users found.
'; return; } allUsers.forEach(user => { const userCard = document.createElement('div'); userCard.className = 'user-card'; userCard.innerHTML = `

${user.name}

Email: ${user.email}

Role: ${user.role}

${user.specialty ? `

Specialty: ${user.specialty}

` : ''}

Status: ${user.status || 'active'}

Joined: ${new Date(user.createdAt || new Date()).toLocaleDateString()}

`; usersList.appendChild(userCard); }); document.querySelectorAll('.edit-user').forEach(button => { button.addEventListener('click', function() { const userId = parseInt(this.getAttribute('data-user-id')); editUser(userId); }); }); document.querySelectorAll('.delete-user').forEach(button => { button.addEventListener('click', function() { const userId = parseInt(this.getAttribute('data-user-id')); deleteUser(userId); }); }); } function loadPatientAssignment() { const allUsers = JSON.parse(localStorage.getItem('allUsers') || '[]'); const patients = allUsers.filter(u => u.role === 'user'); const therapists = allUsers.filter(u => u.role === 'therapist'); const assignPatient = document.getElementById('assign-patient'); const assignTherapist = document.getElementById('assign-therapist'); if (assignPatient) { assignPatient.innerHTML = ''; patients.forEach(patient => { const option = document.createElement('option'); option.value = patient.id; option.textContent = patient.name; assignPatient.appendChild(option); }); } if (assignTherapist) { assignTherapist.innerHTML = ''; therapists.forEach(therapist => { const option = document.createElement('option'); option.value = therapist.id; option.textContent = therapist.name; assignTherapist.appendChild(option); }); } displayCurrentAssignments(); } function displayCurrentAssignments() { const allUsers = JSON.parse(localStorage.getItem('allUsers') || '[]'); const assignments = allUsers.filter(u => u.role === 'user' && u.therapistId); const assignmentsContent = document.getElementById('assignments-content'); if (!assignmentsContent) return; if (assignments.length === 0) { assignmentsContent.innerHTML = '
No patient assignments yet.
'; return; } let assignmentsHTML = ''; assignments.forEach(patient => { const therapist = allUsers.find(u => u.id === patient.therapistId); assignmentsHTML += `

${patient.name}

`; }); assignmentsContent.innerHTML = assignmentsHTML; document.querySelectorAll('.remove-assignment').forEach(button => { button.addEventListener('click', function() { const patientId = parseInt(this.getAttribute('data-patient-id')); removeAssignment(patientId); }); }); } function assignPatient() { const patientId = parseInt(document.getElementById('assign-patient').value); const therapistId = parseInt(document.getElementById('assign-therapist').value); if (!patientId || !therapistId) { alert('Please select both a patient and a therapist'); return; } const allUsers = JSON.parse(localStorage.getItem('allUsers') || '[]'); const patientIndex = allUsers.findIndex(u => u.id === patientId); if (patientIndex !== -1) { allUsers[patientIndex].therapistId = therapistId; localStorage.setItem('allUsers', JSON.stringify(allUsers)); alert('Patient assigned successfully!'); displayCurrentAssignments(); loadAdminStats(); document.getElementById('assign-patient').value = ''; document.getElementById('assign-therapist').value = ''; } } function removeAssignment(patientId) { if (confirm('Are you sure you want to remove this assignment?')) { const allUsers = JSON.parse(localStorage.getItem('allUsers') || '[]'); const patientIndex = allUsers.findIndex(u => u.id === patientId); if (patientIndex !== -1) { allUsers[patientIndex].therapistId = null; localStorage.setItem('allUsers', JSON.stringify(allUsers)); alert('Assignment removed successfully'); displayCurrentAssignments(); loadAdminStats(); } } } function loadSystemReports() { const allUsers = JSON.parse(localStorage.getItem('allUsers') || '[]'); const totalUsers = allUsers.length; const therapists = allUsers.filter(u => u.role === 'therapist').length; const patients = allUsers.filter(u => u.role === 'user').length; const systemReportsContent = document.getElementById('system-reports-content'); if (!systemReportsContent) return; const reportsHTML = `

Platform Overview

Total Users: ${totalUsers}
Therapists: ${therapists}
Patients: ${patients}

Patients: ${patients} (${((patients / totalUsers) * 100).toFixed(1)}%)

Admins: ${totalUsers - therapists - patients} (${(((totalUsers - therapists - patients) / totalUsers) * 100).toFixed(1)}%)

`; systemReportsContent.innerHTML = reportsHTML; } function showAddUserModal() { alert('Add User Modal - This would open a modal to add new users in a real application'); } function editUser(userId) { alert(`Edit User ${userId} - This would open an edit form in a real application`); } function deleteUser(userId) { const currentUser = JSON.parse(localStorage.getItem('currentUser')); if (userId === currentUser.id) { alert('You cannot delete your own account'); return; } if (confirm('Are you sure you want to delete this user?')) { let allUsers = JSON.parse(localStorage.getItem('allUsers') || '[]'); allUsers = allUsers.filter(u => u.id !== userId); localStorage.setItem('allUsers', JSON.stringify(allUsers)); alert('User deleted successfully'); loadUserManagement(); loadAdminStats(); } }
