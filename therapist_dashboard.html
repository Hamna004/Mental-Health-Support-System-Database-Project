`; sessionsList.appendChild(sessionCard); }); }); } function cancelSession(sessionId) { if (confirm('Are you sure you want to cancel this session?')) { cancelSessionAPI(sessionId).then(result => { if (result.success) { alert('Session cancelled successfully'); const currentUser = JSON.parse(localStorage.getItem('currentUser')); loadTherapistsSessions(currentUser); } else { alert('Error: ' + result.message); } }); } } function rescheduleSession(sessionId) { const newDateTime = prompt('Enter new date and time (YYYY-MM-DD HH:MM:SS):'); if (newDateTime) { updateSessionAPI(sessionId, null, newDateTime).then(result => { if (result.success) { alert('Session rescheduled successfully'); const currentUser = JSON.parse(localStorage.getItem('currentUser')); loadTherapistsSessions(currentUser); } else { alert('Error: ' + result.message); } }); } } async function fetchResourcesFromAPI() { try { const response = await fetch(`${API_BASE_URL}resources.php`, { method: 'GET', headers: { 'Content-Type': 'application/json' } }); const data = await response.json(); if (data.success) { return data.resources || []; } else { console.error('Error fetching resources:', data.message); return []; } } catch (error) { console.error('API Error:', error); return []; } } function loadResourcesFromAPI() { const resourcesList = document.getElementById('resources-list'); if (!resourcesList) return; resourcesList.innerHTML = '
